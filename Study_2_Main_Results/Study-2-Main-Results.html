Study 2 — Main (Initial-Session) Analyses

This document reports the primary Study 2 analyses on the initial session (Pre, Stage 1–3, Post). Participants with a Stage-1 baseline (task 0) accuracy of >= 0.90 are excluded by a ceiling-effect screen, consistent with all other Study 2 analyses.

The hypotheses are: H1 learning progression in task performance (accuracy) across stages by labelling condition; H2 confidence, willingness, usefulness, attitude change (Pre → Post), and post-survey experience; and H3 moderation of these effects by individual- difference measures. Baseline equivalence across conditions is checked first.

Manipulation Check: Source Recall

A manipulation check was administered after the learning task, asking participants to recall which type of guidance they had received (AI, AI + Human, or Human). For a source-attribution experiment, this source-recall rate is the key manipulation check: in the AI + Human condition in particular, confusions with the Human or AI labels could dilute the manipulation in ways that bear on interpretation.

Robustness: Item-Level Binomial GLMM

The primary H1 models treat task-level accuracy — a proportion of only 5–6 binary items — as a Gaussian outcome, which can strain linear-model assumptions. As a robustness check we refit the learning-progression effect as an item-level binomial GLMM (the response family used for the early/late analysis), modelling each binary item response directly with crossed participant and item intercepts plus a participant-by-item term that absorbs repeated exposure and binomial overdispersion:

correct ~ Stage * Label + (1 | ID) + (1 | Item) + (1 | ID:Item)

Item responses are read from correction_raw_data.xlsx for the same cohort as the main model (Stage-1 ceiling-screened), so the two describe the same data on different scales. A task-level intercept is omitted (items nest in only three tasks, so its variance sits near the boundary while (1 | Item) already absorbs item clustering); this does not affect the Stage × Label inference. Matching the Gaussian LMM confirms the conclusion is robust to the modelling choice.

Compute Change Scores, Fit Mixed Models

# Pre->Post change per attitude measure: descriptives, change score, and a
# Phase x Label LMM with post-hoc contrasts.
for (msr in measures) {
  cat(paste0("\n===", msr, "===\n"))

  d_sub <- df_long %>%
    filter(measure == msr, Phase %in% c("Pre", "Post"))

  stat_tab_0 <- d_sub %>%
    dplyr::select(ID, Label, Phase, Value, measure) %>%
    group_by(Label, Phase, measure) %>%
    summarise(
      Mean = round(mean(Value, na.rm = TRUE), 2),
      SD = round(sd(Value, na.rm = TRUE), 2),
      N = n(),
      .groups = "drop"
    )

  d_wide <- d_sub %>%
    dplyr::select(ID, Label, Phase, Value) %>%
    pivot_wider(names_from = Phase, values_from = Value) %>%
    mutate(change = Post - Pre)

  stat_tab <- d_wide %>%
    group_by(Label) %>%
    summarise(
      MeanChange = round(mean(change, na.rm = TRUE), 2),
      SD = round(sd(change, na.rm = TRUE), 2),
      N = n(),
      .groups = "drop"
    ) %>%
    mutate(Measure = msr) %>%
    relocate(Measure)

  change_stats[[msr]] <- stat_tab

  d_mod <- d_sub %>%
    filter(!is.na(Value)) %>%
    mutate(
      Phase = factor(Phase, levels = c("Post", "Pre")),
      Phase = relevel(as.factor(Phase), ref = "Post")
    )

  if (nrow(d_mod) > 0) {
    model <- lmer(Value ~ Phase * Label + (1 | ID), data = d_mod)

    aov_tab <- anova(model)
    print(aov_tab)

    eta <- eta_squared(model, partial = TRUE)
    print(eta)
    cat("\n")

    posthoc <- emmeans(model, pairwise ~ Phase)
    cat("\n===t-test: Phase===\n")
    print(posthoc$contrasts)
    cat("\n===Cohen's d: Phase===\n")
    print(eff_size(posthoc, sigma = sigma(model), edf = df.residual(model)))

    posthoc_PL <- emmeans(model, pairwise ~ Phase | Label)
    cat("\n===t-test: Phase | Label===\n")
    print(posthoc_PL$contrasts)
    cat("\n===Cohen's d: Phase | Label===\n")
    print(eff_size(posthoc_PL, sigma = sigma(model), edf = df.residual(model)))

    posthoc_LP <- emmeans(model, pairwise ~ Label | Phase)
    cat("\n===t-test: Label | Phase===\n")
    print(posthoc_LP$contrasts)
    cat("\n===Cohen's d: Label | Phase===\n")
    print(eff_size(posthoc_LP, sigma = sigma(model), edf = df.residual(model)))

    anova_results[[msr]] <- as.data.frame(aov_tab) %>%
      tibble::rownames_to_column("Effect") %>%
      mutate(Measure = msr, .before = 1)

    eta_squared_results[[msr]] <- as.data.frame(eta) %>%
      mutate(Measure = msr, .before = 1)

    posthoc_results[[msr]] <- as.data.frame(posthoc$contrasts) %>%
      mutate(Measure = msr, .before = 1)

    posthoc_results_PL[[msr]] <- as.data.frame(posthoc_PL$contrasts) %>%
      mutate(Measure = msr, .before = 1)

    posthoc_results_LP[[msr]] <- as.data.frame(posthoc_LP$contrasts) %>%
      mutate(Measure = msr, .before = 1)
  }
}

# Loop over every outcome x moderator pair; fit a 3-way moderation LMM each.
for (outcome in outcomes) {
  for (mod in moderators) {
    # Mean-center continuous (pre_) moderators so main effects stay interpretable.
    if (grepl("^pre_", mod)) {
      df_wide[[mod]] <- scale(df_wide[[mod]], scale = FALSE)
    }

    # This outcome only; drop baseline task0 and rows missing the moderator.
    df_task0 <- df_wide %>%
      filter(
        measure == outcome,
        Task != "task0",
        !is.na(.data[[mod]])
      )

    if (nrow(df_task0) == 0) next

    # Stage x Label x moderator, crossed ID/Task random effects.
    formula0 <- as.formula(
      paste0("Value ~ Stage * Label * ", mod, " + (1|ID) + (1|Task) + (1|ID:Task)")
    )

    model0 <- lmer(formula0, data = df_task0)

    anova0 <- anova(model0)
    cat(paste0("\n===", outcome, "+", mod, "===\n"))
    print(anova0)
    # Move term names into a column so they survive the xlsx export.
    ANOVA <- as.data.frame(anova0)
    ANOVA$Variable <- rownames(ANOVA)
    ANOVA <- ANOVA[, c("Variable", setdiff(names(ANOVA), "Variable"))]

    eta0 <- as.data.frame(eta_squared(model0, partial = TRUE))

    coef_tbl <- as.data.frame(summary(model0)$coefficients)
    coef_tbl$Variable <- rownames(coef_tbl)
    coef_tbl <- coef_tbl[, c("Variable", setdiff(names(coef_tbl), "Variable"))]

    posthoc_list <- list()

    # Follow up only the ANOVA terms that reached p < .05.
    sig_terms <- rownames(ANOVA)[which(ANOVA$`Pr(>F)` < 0.05)]

    for (term in sig_terms) {
      clean_term <- gsub("[:/\\?\\*\\[\\]]", "_", term)  # sanitize for list/sheet names
      term_vars <- unlist(strsplit(term, ":"))           # factors making up this term

      # Case 1: term involves a categorical moderator -> estimated means + pairwise.
      if (mod %in% categorical_mods && any(term_vars %in% c(mod, "ModeratorScore", mod))) {
        by_vars <- setdiff(term_vars, c(mod, "ModeratorScore"))
        specs <- c(mod, by_vars)

        em_out <- tryCatch(
          emmeans(model0, specs = specs),
          error = function(e) NULL
        )

        if (!is.null(em_out)) {
          posthoc_list[[clean_term]] <- as.data.frame(pairs(em_out))
        }
      } else if (!(mod %in% categorical_mods) && any(term_vars %in% c(mod, "ModeratorScore", mod))) {
        # Case 2: term involves a continuous moderator -> simple slopes (emtrends).
        other_facs <- setdiff(term_vars, c(mod, "ModeratorScore"))

        # No other factor -> single overall slope; else slope per factor cell.
        if (length(other_facs) == 0) {
          specs <- ~1
        } else {
          specs <- as.formula(paste0("~", paste(other_facs, collapse = "*")))
        }

        et <- tryCatch(
          emtrends(
            model0,
            specs = specs,
            var = ifelse("ModeratorScore" %in% term_vars, "ModeratorScore", mod),
            infer = c(TRUE, TRUE)
          ),
          error = function(e) NULL
        )

        if (!is.null(et)) {
          posthoc_list[[paste0(clean_term, "_trends")]] <-
            as.data.frame(summary(et, infer = c(TRUE, TRUE)))

          # Slopes differ across factor cells -> also test slope differences.
          if (length(other_facs) > 0) {
            ph_pair <- tryCatch(as.data.frame(pairs(et)), error = function(e) NULL)
            if (!is.null(ph_pair)) {
              posthoc_list[[paste0(clean_term, "_trends_pairs")]] <- ph_pair
            }
          }
        }
      } else if (length(term_vars) > 0) {
        # Case 3: term has no moderator (e.g. Stage:Label) -> plain pairwise.
        em_out <- tryCatch(
          emmeans(model0, specs = term_vars),
          error = function(e) NULL
        )

        if (!is.null(em_out)) {
          posthoc_list[[clean_term]] <- as.data.frame(pairs(em_out))
        }
      }
    }

    result_list[[paste0(outcome, "_by_", mod)]] <- list(
      Formula      = as.character(formula0),
      ANOVA        = ANOVA,
      Eta_Squared  = eta0,
      Coef_Summary = coef_tbl,
      Posthoc      = posthoc_list
    )
  }

# Collapse every model's ANOVA into one flat summary table and write it out.
export_sig_effects <- function(
  result_list,
  out_excel_path
) {
  if (!dir.exists(dirname(out_excel_path))) {
    dir.create(dirname(out_excel_path), recursive = TRUE)
  }

  summary_sig_effects <- data.frame(
    DV = character(),
    Moderator = character(),
    Term = character(),
    NumDF = numeric(),
    DenDF = numeric(),
    F_value = numeric(),
    p_value = numeric(),
    eta_p = numeric(),
    stringsAsFactors = FALSE
  )

  for (res_name in names(result_list)) {
    split_name <- unlist(strsplit(res_name, "_by_"))
    DV <- split_name[1]
    Moderator <- split_name[2]

    ANOVA <- result_list[[res_name]]$ANOVA
    Eta <- result_list[[res_name]]$Eta_Squared

    if (is.null(ANOVA)) next
    if (!all(c("Pr(>F)", "F value") %in% names(ANOVA))) next

    # Keep every term: the p-value adjudication step builds the correction
    # family downstream and needs all focal interactions, significant or not.
    sig_rows <- which(ANOVA$`Pr(>F)` <= 1)
    if (length(sig_rows) == 0) next

    for (i in sig_rows) {
      this_term <- ANOVA$Variable[i]
      Ndf <- ANOVA$NumDF[i]
      Ddf <- ANOVA$DenDF[i]
      Fval <- ANOVA$`F value`[i]
      pval <- ANOVA$`Pr(>F)`[i]

      eta_val <- NA
      if (!is.null(Eta) && "Parameter" %in% names(Eta)) {
        eta_row <- Eta[Eta$Parameter == this_term, , drop = FALSE]
        if (nrow(eta_row) == 1) {
          if ("Eta2_partial" %in% names(eta_row)) eta_val <- eta_row$Eta2_partial
          if ("Eta2" %in% names(eta_row) && is.na(eta_val)) eta_val <- eta_row$Eta2
        }
      }

      summary_sig_effects <- rbind(
        summary_sig_effects,
        data.frame(
          DV = DV,
          Moderator = Moderator,
          Term = this_term,
          NumDF = Ndf,
          DenDF = Ddf,
          F_value = Fval,
          p_value = pval,
          eta_p = eta_val,
          stringsAsFactors = FALSE
        )
      )
    }
  }

  summary_sig_effects <- summary_sig_effects %>%
    arrange(DV, Moderator, p_value)

  write.xlsx(summary_sig_effects, file = out_excel_path, rowNames = FALSE)

  cat(sprintf("Significant terms summary saved to %s\n", out_excel_path))

  invisible(summary_sig_effects)
}

  out_excel_path = "results/H3/H3_Within_Task_Moderation_Summary.xlsx"

        filter(Phase == "Pre") %>%
        dplyr::select(ID, BaselineScore = Value)

      df_sub <- df_sub %>%
        left_join(baseline_df, by = "ID") %>%
        mutate(BaselineScore = scale(BaselineScore, scale = FALSE))

      formula_str <- "Value ~ Phase * Label + BaselineScore + (1|ID)"
      model <- lmer(as.formula(formula_str), data = df_sub)
    } else if (mod %in% df_long$measure) {
      # (b) Moderator is another measured scale -> attach its Pre score (centered).
      mod_df <- df_long %>%
        filter(Phase == "Pre", measure == mod) %>%
        dplyr::select(ID, ModeratorScore = Value)

      df_sub <- df_sub %>%
        left_join(mod_df, by = "ID") %>%
        mutate(ModeratorScore = scale(ModeratorScore, scale = FALSE))

      formula_str <- "Value ~ Phase * Label * ModeratorScore + (1|ID)"
      model <- lmer(as.formula(formula_str), data = df_sub)
    } else {
      # (c) Moderator is a fixed demographic column -> use it as-is.
      df_sub <- df_sub %>%
        mutate(ModeratorScore = .data[[mod]])

      formula_str <- "Value ~ Phase * Label * ModeratorScore + (1|ID)"
      model <- lmer(as.formula(formula_str), data = df_sub)
    }

    anova_res <- anova(model)
    cat(paste0("\n===", outcome, "+", mod, "===\n"))
    print(anova_res)
    ANOVA <- as.data.frame(anova_res)
    ANOVA$Variable <- rownames(ANOVA)
    ANOVA <- ANOVA[, c("Variable", setdiff(names(ANOVA), "Variable"))]

    eta_sq <- effectsize::eta_squared(model, partial = TRUE)

    coef_tbl <- as.data.frame(summary(model)$coefficients)
    coef_tbl$Variable <- rownames(coef_tbl)
    coef_tbl <- coef_tbl[, c("Variable", setdiff(names(coef_tbl), "Variable"))]

    sig_terms <- rownames(ANOVA)[which(ANOVA$`Pr(>F)` < 0.05)]

    for (term in sig_terms) {
      clean_term <- gsub("[:/\\?\\*\\[\\]]", "_", term)   # sanitize for list names
      term_vars <- unlist(strsplit(term, ":"))            # factors in this term
      left_facs <- setdiff(term_vars, c(mod, "ModeratorScore"))  # the non-moderator ones

      # Case 1: categorical moderator -> Label contrasts within moderator levels.
      if (mod %in% categorical_mods && any(term_vars %in% c(mod, "ModeratorScore", mod))) {
        ph_em <- tryCatch(
          emmeans(model, pairwise ~ Label | ModeratorScore),
          error = function(e) NULL
        )

        if (!is.null(ph_em)) {
          posthoc_list[[clean_term]] <- as.data.frame(ph_em$contrasts)
        }
      } else if (!(mod %in% categorical_mods) && any(term_vars %in% c(mod, "ModeratorScore", mod))) {
        # Case 2: continuous moderator -> simple slopes, per factor cell if any.
        if (length(left_facs) == 0) {
          specs <- ~1
        } else {
          specs <- as.formula(paste0("~", paste(left_facs, collapse = "*")))
        }

        et <- tryCatch(
          emtrends(
            model,
            specs = specs,
            var = ifelse("ModeratorScore" %in% term_vars, "ModeratorScore", mod),
            infer = c(TRUE, TRUE)
          ),
          error = function(e) NULL
        )

        if (!is.null(et)) {
          posthoc_list[[paste0(clean_term, "_trends")]] <-
            as.data.frame(summary(et, infer = c(TRUE, TRUE)))

          # Also test whether slopes differ across the factor cells.
          if (length(left_facs) > 0) {
            ph_pair <- tryCatch(as.data.frame(pairs(et)), error = function(e) NULL)
            if (!is.null(ph_pair)) {
              posthoc_list[[paste0(clean_term, "_trends_pairs")]] <- ph_pair
            }
          }
        }
      } else if (length(term_vars) > 1) {
        # Case 3a: multi-factor term without moderator (e.g. Phase:Label) -> pairwise.
        em_formula <- as.formula(paste0("~", paste(term_vars, collapse = " * ")))
        em_out <- tryCatch(emmeans(model, em_formula), error = function(e) NULL)

        if (!is.null(em_out)) {
          posthoc_list[[clean_term]] <- as.data.frame(pairs(em_out))
        }
      } else if (length(term_vars) == 1) {
        # Case 3b: single-factor main effect -> pairwise on that factor.
        em_formula <- as.formula(paste0("~", term))
        em_out <- tryCatch(emmeans(model, em_formula), error = function(e) NULL)

        if (!is.null(em_out)) {
          posthoc_list[[clean_term]] <- as.data.frame(pairs(em_out))
        }
      }
    }

    result_list[[paste0(outcome, "_by_", mod)]] <- list(
      Formula      = formula_str,
      ANOVA        = ANOVA,
      Eta_Squared  = as.data.frame(eta_sq),
      Coef_Summary = coef_tbl,
      Posthoc      = posthoc_list
    )
  }
}

# Same as above (keeps every term; focal filtering happens in p-value adjudication).
export_sig_effects <- function(result_list, out_excel_path) {
  if (!dir.exists(dirname(out_excel_path))) {
    dir.create(dirname(out_excel_path), recursive = TRUE)
  }

  summary_sig_effects <- data.frame(
    DV = character(),
    Moderator = character(),
    Term = character(),
    NumDF = numeric(),
    DenDF = numeric(),
    F_value = numeric(),
    p_value = numeric(),
    eta_p = numeric(),
    stringsAsFactors = FALSE
  )

  for (res_name in names(result_list)) {
    split_name <- unlist(strsplit(res_name, "_by_"))
    DV <- split_name[1]
    Moderator <- split_name[2]

    ANOVA <- result_list[[res_name]]$ANOVA
    Eta <- result_list[[res_name]]$Eta_Squared

    if (is.null(ANOVA)) next
    if (!all(c("Pr(>F)", "F value") %in% names(ANOVA))) next

    # Keep every term (see note in the within-task version above): the adjudication
    # step needs the full set of focal interactions to build the correction family.
    sig_rows <- which(ANOVA$`Pr(>F)` <= 1)
    if (length(sig_rows) == 0) next

    for (i in sig_rows) {
      this_term <- ANOVA$Variable[i]
      Ndf <- ANOVA$NumDF[i]
      Ddf <- ANOVA$DenDF[i]
      Fval <- ANOVA$`F value`[i]
      pval <- ANOVA$`Pr(>F)`[i]

      eta_val <- NA
      if (!is.null(Eta) && "Parameter" %in% names(Eta)) {
        eta_row <- Eta[Eta$Parameter == this_term, , drop = FALSE]
        if (nrow(eta_row) == 1) {
          if ("Eta2_partial" %in% names(eta_row)) eta_val <- eta_row$Eta2_partial
          if ("Eta2" %in% names(eta_row) && is.na(eta_val)) eta_val <- eta_row$Eta2
        }
      }

      summary_sig_effects <- rbind(
        summary_sig_effects,
        data.frame(
          DV = DV,
          Moderator = Moderator,
          Term = this_term,
          NumDF = Ndf,
          DenDF = Ddf,
          F_value = Fval,
          p_value = pval,
          eta_p = eta_val,
          stringsAsFactors = FALSE
        )
      )
    }
  }

  summary_sig_effects <- summary_sig_effects %>%
    dplyr::arrange(DV, Moderator, p_value)

  write.xlsx(summary_sig_effects, file = out_excel_path, rowNames = FALSE)

  cat(sprintf("Significant terms summary saved to %s\n", out_excel_path))

  invisible(summary_sig_effects)
}

# Each categorical moderator x 3 post-survey outcomes (learning, satisfaction, sign-up).
for (mod in moderators) {
  # Outcome 1: perceived learning (ANOVA).
  formula_anova <- as.formula(paste0("Stats_Improvement ~ Label * ", mod))
  cat("\n===", "Stats Improvement +", mod, "===\n")
  anova_mod <- aov(formula_anova, data = df)
  print(summary(anova_mod))

  anova_tbl <- as.data.frame(summary(anova_mod)[[1]])
  anova_tbl$Variable <- rownames(anova_tbl)
  anova_tbl <- anova_tbl[, c("Variable", setdiff(names(anova_tbl), "Variable"))]

  means_learning_mod <- df %>%
    group_by(Label, !!sym(mod)) %>%
    summarise(
      Mean = round(mean(Stats_Improvement, na.rm = TRUE), 2),
      SD = round(sd(Stats_Improvement, na.rm = TRUE), 2),
      N = n(),
      .groups = "drop"
    )

  emm_formula_interaction <- as.formula(paste("pairwise ~ Label *", mod))

  posthoc_learning_mod <- tryCatch(
    as.data.frame(emmeans(anova_mod, emm_formula_interaction)),
    error = function(e) data.frame()
  )

  # Outcome 2: satisfaction (ordinal logistic / polr).
  formula_satis <- as.formula(paste0("Satisfaction ~ Label * ", mod))
  cat("\n===", "Satisfaction +", mod, "===\n")
  satis_model_mod <- MASS::polr(formula_satis, data = df, Hess = TRUE)

  satis_aov <- Anova(satis_model_mod)
  print(satis_aov)

  # polr gives no p-values; derive them from the t (Wald z) values.
  coefs <- summary(satis_model_mod)$coefficients
  pvals <- 2 * pnorm(-abs(coefs[, "t value"]))
  Satisfaction_Coefficients <- as.data.frame(cbind(coefs, "p value" = pvals))

    group_by(Label, !!sym(mod)) %>%

      N = n(),
      .groups = "drop"
    )

    as.data.frame(emmeans(satis_model_mod, emm_formula_interaction)),
    error = function(e) data.frame()
  )

    SignUp_Logit_Coeffs = as.data.frame(summary(logit_model_mod)$coefficients),
